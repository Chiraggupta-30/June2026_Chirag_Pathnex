- name: Setup Docker Container for Nginx
  hosts: all
  become: yes 
  tasks:
   - name: Pull Nginx image
     docker_image:
       name: nginx
       source: pull
   - name: Run Nginx Container 
     docker_container:
       name: nginx-container
       image: nginx
       state: started 
       published_ports:
          - "8080:80"

apiVersion: apps/v1
kind: Deployment 
metadata: 
  name:  pathnex-deployment 
 spec:
    replicas: 1
    selector:
       matchLabels:
         app: pathnex-app
      template:
         metadata: 
            labels: 
                app: pathnex-app
              spec:
                 containers:
                     -name: nginx
                      image: nginx
                      ports: 
                         - containerPort: 80

---
