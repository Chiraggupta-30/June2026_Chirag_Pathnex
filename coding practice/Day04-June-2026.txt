##  terraform - EC2 with Key Pair (c61.8xlarge)
    
   provider "aws" { 
      region = "us-east-1"
  } 

  resource "aws_key_pair" "PathnexKey" {
    key_name = "PathnexKey"
    public_key = "ssh-rsa AAA...."
  } 

 resource "aws_instance1 "PathnexEC2" { 
   ami            = "ami-0abcd1234abcd1234"
   instance_type  = "c61.8xlarge"
   key_name       =  aws_key_pair.PathnexKey.Key_name
   
  tags = {  
    Name = "Pathnex-EC2"
   }
}

# Jenkins Pipeline - Parallel Build & Test  

  pipeline { 
       agent any 
       stages { 
          stage('Parallel Tasks') {
            parallel {  
               stage('Build') {  
                  steps  { 
                      sh 'mcn clean compile'
                  }
               }
                stage('Test') {
                    steps {
                       sh 'mvn test'
                     }
                  }
              }
          }
       }
     }




## Gitlab CI - Parallel Jobs 
 

  stages:
     -build-test 
  

   maven-build 
      stage: build-test 
      image: maven:3.8.1.jdk-17
      script:
         - git clone https://github.com/Pathnex/sample-java-app.git
         - cd sample-java-app 
         - mvn clean compile 
       parallel: 1


   maven-test: 
    stage: build-test 
    image: maven:3.8.1.-jdk-17
    script:
      - cd sample-java-app
      - mvn test 
     parallel: 1
