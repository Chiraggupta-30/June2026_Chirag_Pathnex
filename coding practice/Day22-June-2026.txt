# Ansible code


---
- name: Install MariaDB for ChiragServer
  hosts: all
  become: yes 
  
  tasks: 
    - name: Install MariaDB
      yum: 
        name: mariadb-server
        state: present 
    
    - name: Start MariaDB
      service: 
        name: mariadb
        state: started 
        enabled: yes

# Jenkins 


pipeline { 
     agent any 
     environment { 
           INSTITUTE_NAME = "ChiragServer" 
   } 
   stages { 
        stage('Checkout') { 
            steps {
               git url: 'https://github.com/ChiragServer/sample-java-app.git'
          }
      }
      stage('Build' { 
           steps { 
               sh 'mvn clean package'
            } 
         } 
      }
      post { 
          success { 
              mail to: 'supportdevops@chirag.com',
               subject: "SUCCESS: Build #${env.BUILD_NUMBER}",
               body:    "Build completed successfully for $INSTITUTE_NAME"
         }  
         failure { 
             mail to: 'supportdevops@chirag.com',
               subject: "FAILURE: Build #${env.BUILD_NUMBER}",
               body:    "Build failed for $INSTITUTE_NAME"
         }
      } 
   }
