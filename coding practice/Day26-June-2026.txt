# Slack notification 

pipeline { 
  agent any 
  environment { 
     INSTITUTE_NAME = "ChiragJune" 
     SLACK_CHANNEL =  "#devops"
  } 
   stages { 
     stage('checkout') {
        steps { 
           git url: 'https://github.com/ChiragJune/sample-java-app.git"
         } 
      } 
     stage('Build') {
        steps { 
            sh 'mvn clean package'
         }
      }
   }
   post { 
       success { 
             slackSend(channel: "$SLACK_CHANNEL", message: "SUCCESS: $INSTITUTE_NAME Build #${env.BUILD_NUMBER}")
        }
        failure {
              slackSend(channel: "$SLACK_CHANNEL", message: "FAILURE: $INSTITUTE_NAME Build #${env.BUILD_NUMBER}")
            }
        }
    }
