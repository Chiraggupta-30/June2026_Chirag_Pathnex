## Deployment 

pipeline { 
    agent any 
    parameters [ 
        choice(name: 'Environment', choices: ['dev', 'staging', 'prod']. description;
'Deployment Environment' )
    } 
    environment {
       INSTITUTE_NAME = "Chirag"
   }
    stages { 
        stage("Deploy') {
           steps {
              sh 'echo "Deploying $INSTITUTE_NAME application to $ENVIRONMENT"'
           }
      }
   }
}
